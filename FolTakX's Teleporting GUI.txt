-- Scripts:

local function TeleportPlayerScript() -- TeleportUser.Script 
	local script = Instance.new('Script', TeleportUser)

	script.Parent.MouseButton1Click:Connect(function()
		game.Players.LocalPlayer.Character:MoveTo(game.Workspace[script.Parent.Parent.Text.Name].Head.Position)
	end)
end
local function CloseGuiScript() -- Close.Script 
	local script = Instance.new('Script', Close)

	script.Parent.MouseButton1Click:Connect(function()
		script.Parent.Parent.Visible = false
		script.Parent.Parent.Parent.Open.Visible = true
	end)
end
local function OpenGuiScript() -- Open.Script 
	local script = Instance.new('Script', Open)

	script.Parent.MouseButton1Click:Connect(function()
		script.Parent.Visible = false
		script.Parent.Parent.Frame.Visible = true
	end)
end
